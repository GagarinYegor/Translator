begin
    A, B, C: integer;  {Комментарий}
    A := 0;
    read A;
    loop
    begin
        A := A - 1;
        C := 0;
        loop
        begin
            write C;
            C := C + 1;
            if C > 1 then goto innerCycleExitLabel;
        end;
        end;
        innerCycleExitLabel:
        if A <= 0 then goto exitLabel;
    end;
    end;
    exitLabel:
    write A;
    B := 0;
    read B;
    cycleLabel:
    write B;
    B := B - 1;
    if B > 0 then goto cycleLabel else goto cycleExitLabel;
    cycleExitLabel:;
end.
