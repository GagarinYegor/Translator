{Пример программы}
begin
    A := 0;
    read A;
    loop
        A := A - 1;
        if A <= 0 then goto exitLabel;
    end;
    exitLabel:
    write A;

    B := 0;
    read B;
    cycleLabel:
    B := B - 1;
    if B > 0 then goto cycleLabel else goto cycleExitLabel;
    cycleExitLabel:
    write B;
end.
